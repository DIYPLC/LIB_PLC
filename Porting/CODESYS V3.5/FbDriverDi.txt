FUNCTION_BLOCK FbDriverDi //Драйвер дискретного входа.

VAR_INPUT //Входные переменные, сохраняемые.
In       :BOOL := FALSE; //Дискретный вход ПЛК.
SimValue :BOOL := FALSE; //Значение для Out при симуляции.
SimOn    :BOOL := FALSE; //Включить симуляцию.
END_VAR

VAR_OUTPUT //Выходные переменные, сохраняемые.
Out :BOOL := FALSE; //Выход дискретного сигнала.
END_VAR

//          FbDriverDi
//    +--------------------+
//    |     FbDriverDi     |
// ->-|In               Out|->-
//   -|SimValue            |
//   -|SimOn               |
//    +--------------------+

IF (SimOn) THEN
//Симуляция включена.
//Режим используется для отладки и при нештатной работе.
Out := SimValue;
ELSE
//Симуляция выключена.
//Штатный режим работы.
Out := In;
END_IF;
