PROGRAM PLC_PRG

VAR
DbTask100ms : FbTask100ms;
DbTaskCFC : FbTaskCFC;
DbSampleTime : FbSampleTime;
END_VAR

DbSampleTime();

GV.Ts :=  DbSampleTime.Ts;

DbTask100ms.Ts_ms := GV.Ts_ms;

DbTask100ms();

DbTaskCFC.Ts_ms := GV.Ts_ms;
DbTaskCFC.Reset := GV.Reset;
DbTaskCFC();

IF (GV.Reset) THEN
GV.Reset := FALSE;
END_IF
